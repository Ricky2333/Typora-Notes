	printf("this is a new thread, thread ID = %d\n", newthid);
	return NULL; 
} 

int main(void) {
	pthread_t thid
	printf("main thread ,ID is %d\n",pthread_self()); 
	if(pthread_create(&thid, NULL, (void *)thread, NULL) != 0) {
